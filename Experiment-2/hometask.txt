Q1---------------------
SELECT candidate_id
FROM candidates
WHERE skill IN ('Python', 'Tableau', 'PostgreSQL')
GROUP BY candidate_id
HAVING COUNT(skill) = 3
ORDER BY candidate_id ASC;

Q2---------------------
SELECT p.page_id
FROM pages p
LEFT JOIN page_likes l 
  ON p.page_id = l.page_id
WHERE l.page_id IS NULL
ORDER BY p.page_id ASC;
